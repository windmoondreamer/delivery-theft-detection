/**
  * @brief System Clock Configuration
  * @retval None
  */
int IIR_Filter(int filterInput) {
	float filterInputF = (float)filterInput;
	float filterOutputNum = 0.0;
	float filterOutputDen = 0.0;
	float filterOutput = 0.0;

	//shift input/output buffer
	for (int i = FILTER_LENGTH-1 ; i > 0 ; i--) {
		inputBuf[i] = inputBuf[i-1];
		outputBuf[i] = outputBuf[i-1];
	}

	//place new input
	inputBuf[0] = filterInputF;
	outputBuf[0] = 0.0;

	//convolution sum
	for (int i = 0 ; i < FILTER_LENGTH ; i++) {
		filterOutputNum += (inputBuf[i]*filter_num_coeffs[i]);
		filterOutputDen += (outputBuf[i]*filter_den_coeffs[i]);
	}

	//output calculation
	filterOutput = filterOutputNum - filterOutputDen;

	//place new output
	outputBuf[0] = filterOutput;

	return (int) filterOutput;
}
